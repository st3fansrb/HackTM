int get daysUntilExpiry {
  // Firestore Timestamps are UTC; normalize both to UTC to prevent timezone drift
  final expiryUtc = expiryDate.toDate().toUtc();
  final nowUtc = DateTime.now().toUtc();

  final diffDays = expiryUtc.difference(nowUtc).inDays;

  // Clamp to 0 to avoid negative values for already expired items
  return diffDays > 0 ? diffDays : 0;
}
