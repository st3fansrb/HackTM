import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';

// Exemplu de provider pentru AppTheme
final appThemeProvider = Provider((ref) => AppTheme());

class AppTheme {
  final Color primaryColor = const Color(0xFF2ECC71);
  final String fontFamily = 'Nunito';
}

// Widget Text cu stilul AppTheme
class StyledText extends ConsumerWidget {
  final String text;

  const StyledText({required this.text});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final theme = ref.watch(appThemeProvider);
    return Text(
      text,
      style: TextStyle(
        fontFamily: theme.fontFamily,
        color: theme.primaryColor,
        fontSize: 16, // Poți ajusta dimensiunea fontului aici
      ),
    );
  }
}
